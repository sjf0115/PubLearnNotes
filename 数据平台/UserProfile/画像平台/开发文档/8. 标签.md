



标签宽表协议：
```sql
CREATE TABLE IF NOT EXISTS `user_profile_label`(
  `user_id` bigint COMMENT '用户ID',
  `gender` string COMMENT '性别',
  `brand` string COMMENT '品牌',
  `prov` string COMMENT '常驻省份',
  ...
)
COMMENT '画像标签宽表'
PARTITIONED BY (`dt` string COMMENT '日期-yyyyMMdd');
```

> 标签ID -> 标签宽表数据集 -> 物理表字段

标签Bitmap表协议：
```sql
CREATE TABLE IF NOT EXISTS `user_profile_bitmap_label`(
  `bucket_id` bigint COMMENT '标签分桶',
  `label_name` string COMMENT '标签名称：例如，性别',
  `label_value` string COMMENT '标签值：例如，男',
  `label_bitmap` binary COMMENT '标签用户Bitmap'
)
COMMENT '画像标签Bitmap表'
PARTITIONED BY (`dt` string COMMENT '日期-yyyyMMdd');
```
> 标签ID -> 标签Bitmap数据集 -> label_name = 'xxx'

标签BSI表协议：
```sql
CREATE TABLE IF NOT EXISTS `user_profile_bsi_label`(
  `bucket_id` bigint COMMENT '标签分桶',
  `label_name` string COMMENT '标签名称：例如，积分',
  `label_bsi` binary COMMENT '标签BSI值'
)
COMMENT '画像标签BSI表'
PARTITIONED BY (`dt` string COMMENT '日期-yyyyMMdd');
```
> 标签ID -> 标签BSI数据集 -> label_name = 'xxx'
